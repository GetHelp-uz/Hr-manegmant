DATABASE SQL STRUCTURE (POSTGRESQL)

CREATE TABLE companies (
id SERIAL PRIMARY KEY,
name VARCHAR(255),
phone VARCHAR(50),
email VARCHAR(255),
password TEXT,
logo TEXT,
subscription_plan VARCHAR(50),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE admins (
id SERIAL PRIMARY KEY,
company_id INTEGER REFERENCES companies(id),
name VARCHAR(255),
email VARCHAR(255),
password TEXT,
role VARCHAR(50),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE employees (
id SERIAL PRIMARY KEY,
company_id INTEGER REFERENCES companies(id),
full_name VARCHAR(255),
phone VARCHAR(50),
position VARCHAR(255),
salary_type VARCHAR(50),
hourly_rate NUMERIC,
monthly_salary NUMERIC,
qr_code TEXT,
telegram_id VARCHAR(100),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE devices (
id SERIAL PRIMARY KEY,
company_id INTEGER REFERENCES companies(id),
device_name VARCHAR(255),
location VARCHAR(255),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE photos (
id SERIAL PRIMARY KEY,
employee_id INTEGER REFERENCES employees(id),
company_id INTEGER REFERENCES companies(id),
image_url TEXT,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE attendance (
id SERIAL PRIMARY KEY,
employee_id INTEGER REFERENCES employees(id),
company_id INTEGER REFERENCES companies(id),
check_in TIMESTAMP,
check_out TIMESTAMP,
work_hours NUMERIC,
photo_id INTEGER REFERENCES photos(id),
device_id INTEGER REFERENCES devices(id),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE subscriptions (
id SERIAL PRIMARY KEY,
company_id INTEGER REFERENCES companies(id),
plan_name VARCHAR(50),
employee_limit INTEGER,
price NUMERIC,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

QR CODE GENERATOR (NODE.JS)

const QRCode = require("qrcode");

async function generateEmployeeQR(employeeId, companyId){

const qrData = JSON.stringify({
employee_id: employeeId,
company_id: companyId
});

const qr = await QRCode.toDataURL(qrData);

return qr;
}

module.exports = generateEmployeeQR;

QR SCAN API (EXPRESS BACKEND)

app.post("/scan", async(req,res)=>{

const {employee_id,company_id,device_id}=req.body;

const now=new Date();

const attendance=await db.query(
"SELECT * FROM attendance WHERE employee_id=$1 AND DATE(created_at)=CURRENT_DATE",
[employee_id]
);

if(attendance.rows.length===0){

await db.query(
"INSERT INTO attendance (employee_id,company_id,check_in,device_id) VALUES ($1,$2,$3,$4)",
[employee_id,company_id,now,device_id]
);

res.json({message:"CHECK IN"});

}else{

await db.query(
"UPDATE attendance SET check_out=$1 WHERE id=$2",
[now,attendance.rows[0].id]
);

res.json({message:"CHECK OUT"});
}

});
